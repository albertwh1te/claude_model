    69→    def __init__(self, opts: EngineClientOpts):
    70→        """
    71→        Initialize EngineQueryClient with provided options.
    72→
    73→        Args:
    74→            opts (EngineClientOpts): Options for the client.
    75→        """
    76→        self._opts: EngineClientOpts = EngineClientOpts.parse_obj(opts)
    77→        self.url: str = self._opts.url
    78→        self.url_v2: str = self.url.replace("/v1", "") + "/v2"
    79→        self.session = requests.Session()  # type: ignore
    80→
    81→    def query(self, req: QueryRequest) -> QueryResponse:
    82→        """
    83→        Send a query to the engine.
    84→
    85→        Args:
    86→            req (QueryRequest): The query request parameters.
    87→
    88→        Returns:
    89→            QueryResponse: The response from the engine.
    90→
    91→        Raises:
    92→            BadStatusCodeException: If the response status code is not 200.
    93→            QueryFailedException: If the query status is not "success".
    94→        """
    95→        res = self.session.post(f"{self.url}/query", json=req.dict())
    96→        if res.status_code != 200:
    97→            raise BadStatusCodeException(res.text)
    98→        try:
    99→            query_res = QueryResponse(**res.json())
   100→        except Exception:
   101→            raise QueryFailedException(res.text)
   102→        if query_res.status != "success":
   103→            raise QueryFailedException(res.text)
   104→        return query_res
   105→
   106→    def _query_v2(self, url):
   107→        res = self.session.get(url)
   108→        if res.status_code != 200:
   109→            raise Exception(res.text)
   110→        return res.json()

   119→        res = self.session.get(f"{self.url}/symbols?")
   120→        if res.status_code != 200:

   185→    def get_subaccount_info(
   186→        self,
   187→        subaccount: str,

   189→        pre_state: Optional[bool] = None,
   190→    ) -> SubaccountInfoData:
   191→        """
   192→        Query the engine for the state of a subaccount, including balances.
   193→
   194→        Args:
   195→            subaccount (str): Identifier of the subaccount (owner's address + subaccount name) sent as a hex string.

   214→        return ensure_data_type(
   215→            self.query(
   216→                QuerySubaccountInfoParams(
   217→                    subaccount=subaccount, txns=txns_str, pre_state=pre_state_str

   461→    def _get_subaccount_product_position(
   462→        self, subaccount: str, product_id: int
   463→    ) -> SubaccountPosition:
   464→        summary = self.get_subaccount_info(subaccount)
   465→        try:
   466→            balance = [
   467→                balance
   468→                for balance in summary.spot_balances + summary.perp_balances
   469→                if balance.product_id == product_id
   470→            ][0]
   471→            product = [
   472→                product
   473→                for product in summary.spot_products + summary.perp_products
   474→                if product.product_id == product_id
   475→            ][0]
   476→        except Exception as e:
   477→            raise Exception(f"Invalid product id provided {product_id}. Error: {e}")
   478→        return SubaccountPosition(balance=balance, product=product)
