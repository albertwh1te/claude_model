    43→    """
    44→    Client class for executing operations against the off-chain engine.

    47→    def __init__(
    48→        self, opts: EngineClientOpts, querier: Optional[EngineQueryClient] = None
    49→    ):
    50→        """
    51→        Initialize the EngineExecuteClient with provided options.
    52→
    53→        Args:
    54→            opts (EngineClientOpts): Options for the client.
    55→
    56→            querier (EngineQueryClient, optional): An EngineQueryClient instance. If not provided, a new one is created.
    57→        """
    58→        super().__init__(opts)
    59→        self._querier = querier or EngineQueryClient(opts)

    61→        self.url: str = self._opts.url
    62→        self.session = requests.Session()

    64→    def tx_nonce(self, sender: str) -> int:

    73→    @singledispatchmethod
    74→    def execute(self, params: Union[ExecuteParams, ExecuteRequest]) -> ExecuteResponse:
    75→        """
    76→        Executes the operation defined by the provided parameters.
    77→
    78→        Args:
    79→            params (ExecuteParams): The parameters for the operation to execute. This can represent a variety of operations, such as placing orders, cancelling orders, and more.
    80→
    81→        Returns:
    82→            ExecuteResponse: The response from the executed operation.
    83→        """
    84→        req: ExecuteRequest = (
    85→            params if is_instance_of_union(params, ExecuteRequest) else to_execute_request(params)  # type: ignore
    86→        )
    87→        return self._execute(req)
    88→
    89→    @execute.register
    90→    def _(self, req: dict) -> ExecuteResponse:
    91→        """
    92→        Overloaded method to execute the operation defined by the provided request.
    93→
    94→        Args:
    95→            req (dict): The request data for the operation to execute. Can be a dictionary or an instance of ExecuteRequest.
    96→
    97→        Returns:
    98→            ExecuteResponse: The response from the executed operation.
    99→        """
   100→        parsed_req: ExecuteRequest = NadoBaseModel.parse_obj(req)  # type: ignore
   101→        return self._execute(parsed_req)

   103→    def _execute(self, req: ExecuteRequest) -> ExecuteResponse:
   104→        """
   105→        Internal method to execute the operation. Sends request to the server.
   106→
   107→        Args:
   108→            req (ExecuteRequest): The request data for the operation to execute.
   109→
   110→        Returns:
   111→            ExecuteResponse: The response from the executed operation.
   112→
   113→        Raises:
   114→            BadStatusCodeException: If the server response status code is not 200.
   115→            ExecuteFailedException: If there's an error in the execution or the response status is not "success".
   116→        """
   117→        res = self.session.post(f"{self.url}/execute", json=req.dict())
   118→        if res.status_code != 200:
   119→            raise BadStatusCodeException(res.text)
   120→        try:
   121→            execute_res = ExecuteResponse(**res.json(), req=req.dict())
   122→        except Exception:
   123→            raise ExecuteFailedException(res.text)
   124→        if execute_res.status != "success":
   125→            raise ExecuteFailedException(res.text)
   126→        return execute_res

   128→    def _assert_book_not_empty(
   129→        self, bids: list[MarketLiquidity], asks: list[MarketLiquidity], is_bid: bool
   130→    ):
   131→        book_is_empty = (is_bid and len(bids) == 0) or (not is_bid and len(asks) == 0)
   132→        if book_is_empty:
   133→            raise Exception("Orderbook is empty.")

   135→    def place_order(self, params: PlaceOrderParams) -> ExecuteResponse:
   136→        """
   137→        Execute a place order operation.
   138→
   139→        Args:
   140→            params (PlaceOrderParams): Parameters required for placing an order.
   141→            The parameters include the order details and the product_id.
   142→
   143→        Returns:
   144→            ExecuteResponse: Response of the execution, including status and potential error message.
   145→        """
   146→        params = PlaceOrderParams.parse_obj(params)
   147→        params.order = self.prepare_execute_params(params.order, True)
   148→        params.signature = params.signature or self._sign(
   149→            NadoExecuteType.PLACE_ORDER, params.order.dict(), params.product_id
   150→        )
   151→        return self.execute(params)

   153→    def place_market_order(self, params: PlaceMarketOrderParams) -> ExecuteResponse:
   154→        """
   155→        Places an FOK order using top of the book price with provided slippage.
   156→
   157→        Args:
   158→            params (PlaceMarketOrderParams): Parameters required for placing a market order.
   159→
   160→        Returns:
   161→            ExecuteResponse: Response of the execution, including status and potential error message.
   162→        """
   163→        orderbook = self._querier.get_market_liquidity(params.product_id, 1)
   164→        is_bid = int(params.market_order.amount) > 0
   165→        self._assert_book_not_empty(orderbook.bids, orderbook.asks, is_bid)
   166→        slippage = to_x18(params.slippage or 0.005)  # defaults to 0.5%
   167→        market_price_x18 = (
   168→            mul_x18(orderbook.bids[0][0], to_x18(1) + slippage)
   169→            if is_bid
   170→            else mul_x18(orderbook.asks[0][0], to_x18(1) - slippage)
   171→        )
   172→        price_increment_x18 = self._querier._get_subaccount_product_position(
   173→            subaccount_to_hex(params.market_order.sender), params.product_id
   174→        ).product.book_info.price_increment_x18
   175→        order = OrderParams(
   176→            sender=params.market_order.sender,
   177→            amount=params.market_order.amount,
   178→            nonce=params.market_order.nonce,
   179→            priceX18=round_x18(market_price_x18, price_increment_x18),
   180→            expiration=get_expiration_timestamp(1000),
   181→            appendix=build_appendix(
   182→                OrderType.FOK, reduce_only=bool(params.reduce_only)
   183→            ),
   184→        )
   185→        return self.place_order(
   186→            PlaceOrderParams(  # type: ignore
   187→                product_id=params.product_id,
   188→                order=order,
   189→                spot_leverage=params.spot_leverage,
   190→                signature=params.signature,
   191→            )
   192→        )

   354→    def close_position(
   355→        self, subaccount: Subaccount, product_id: int
   356→    ) -> ExecuteResponse:
   357→        """
   358→        Execute a place order operation to close a position for the provided `product_id`.
   359→
   360→        Attributes:
   361→            subaccount (Subaccount): The subaccount to close position for.
   362→            product_id (int): The ID of the product to close position for.
   363→
   364→        Returns:
   365→            ExecuteResponse: Response of the execution, including status and potential error message.
   366→        """
   367→        subaccount = subaccount_to_hex(subaccount)
   368→        position = self._querier._get_subaccount_product_position(
   369→            subaccount, product_id
   370→        )
   371→        balance, product = position.balance, position.product
   372→        closing_spread_x18 = to_x18(0.005)
   373→        closing_price_x18 = (
   374→            mul_x18(product.oracle_price_x18, to_x18(1) - closing_spread_x18)
   375→            if int(balance.balance.amount) > 0
   376→            else mul_x18(product.oracle_price_x18, to_x18(1) + closing_spread_x18)
   377→        )
   378→        return self.place_order(
   379→            PlaceOrderParams(  # type: ignore
   380→                product_id=product_id,
   381→                order=OrderParams(  # type: ignore
   382→                    sender=subaccount,
   383→                    amount=-round_x18(
   384→                        balance.balance.amount,
   385→                        product.book_info.size_increment,
   386→                    ),
   387→                    priceX18=round_x18(
   388→                        closing_price_x18,
   389→                        product.book_info.price_increment_x18,
   390→                    ),
   391→                    expiration=get_expiration_timestamp(1000),
   392→                    appendix=build_appendix(OrderType.FOK, reduce_only=True),
   393→                ),
   394→            )
   395→        )
