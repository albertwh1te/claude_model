   100→    def _query(self, req: IndexerRequest) -> IndexerResponse:
   101→        res = self.session.post(self.url, json=req.dict())
   102→        if res.status_code != 200:
   103→            raise Exception(res.text)
   104→        try:
   105→            indexer_res = IndexerResponse(data=res.json())
   106→        except Exception:
   107→            raise Exception(res.text)
   108→        return indexer_res
