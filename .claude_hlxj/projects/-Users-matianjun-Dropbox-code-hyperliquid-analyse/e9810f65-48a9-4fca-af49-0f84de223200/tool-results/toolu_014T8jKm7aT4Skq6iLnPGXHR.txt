   340→@app.command("generate_weights_4h")
   341→def generate_weights_4h(
   342→    prediction_file: str = "regression_model_predictions.feather",
   343→    output_file: str = "ranker_weight_4h.feather",
   344→):
   345→    """
   346→    4H版本：根据模型预测结果生成投资组合权重
   347→
   348→    Args:
   349→        prediction_file: 预测结果文件 (来自 tune_regression_predict_4h)
   350→        output_file: 输出的权重文件
   351→    """
   352→    logger.info(f"[4H] Loading predictions from {prediction_file}")
   353→
   354→    ranker_probability = pd.read_feather(prediction_file)
   355→    ranker_probability["time"] = pd.to_datetime(ranker_probability["time"])
   356→
   357→    logger.info(
   358→        f"[4H] Loaded {len(ranker_probability)} predictions, "
   359→        f"from {ranker_probability['time'].min()} to {ranker_probability['time'].max()}"
   360→    )
   361→
   362→    # 使用 4H 版本的权重生成器
   363→    weight_generator = RegressionPortfolioOptimizerGenerator4H()
   364→
   365→    logger.info(
   366→        f"[4H] Generating weights using RegressionPortfolioOptimizerGenerator4H"
   367→    )
   368→
   369→    ranker_weight = weight_generator.generate_weights(
   370→        ranker_probability=ranker_probability,
   371→        model_file_path=None,  # 不需要模型文件
   372→        save_path=output_file,
   373→        end_time=None,
   374→    )
   375→
   376→    logger.info(f"[4H] Weights saved to {output_file}")
   377→    logger.info(f"[4H] Generated {len(ranker_weight)} weight records")
   378→
   379→    return ranker_weight

   419→@app.command("pipeline_4h")
   420→@notify_me_command("finished 4H full pipeline")
   421→def pipeline_4h(
   422→    skip_feature: bool = False,
   423→    skip_tune: bool = False,
   424→    tune_trials: int = 100,
   425→):
   426→    """
   427→    4H版本：完整的一键流水线
   428→
   429→    流程:
   430→    1. 生成4H特征 (feature_4h)
   431→    2. 调优并训练模型 (tune_regression_predict_4h)
   432→    3. 生成权重 (generate_weights_4h)
   433→    4. 运行回测 (backtest_4h)
   434→
   435→    Args:
   436→        skip_feature: 跳过特征生成 (如果已有特征文件)
   437→        skip_tune: 跳过调优 (使用默认参数)
   438→        tune_trials: Optuna 调优次数
   439→    """
   440→    logger.info("=" * 60)
   441→    logger.info("[4H Pipeline] Starting full 4H prediction pipeline")
   442→    logger.info("=" * 60)
   443→
   444→    # Step 1: 生成特征
   445→    if not skip_feature:
   446→        logger.info("[4H Pipeline] Step 1/4: Generating features...")
   447→        feature_4h()
   448→    else:
   449→        logger.info("[4H Pipeline] Step 1/4: Skipping feature generation")
   450→
   451→    # Step 2: 调优并训练模型
   452→    if not skip_tune:
   453→        logger.info(
   454→            f"[4H Pipeline] Step 2/4: Tuning and training (trials={tune_trials})..."
   455→        )
   456→        tune_regression_predict_4h(times=tune_trials)
   457→    else:
   458→        logger.info("[4H Pipeline] Step 2/4: Skipping tuning, using default params")
   459→        # 使用默认参数训练
   460→        symbol_list = get_symbols_by_name("full")
   461→        feature_file_name = "tmp_feature_4h_full.csv"
   462→        target_column = "target_4hour_zscore"
   463→
   464→        RegressionModelTrain(
   465→            hyper_parameters=RANKER_REGRESSION_BASE_PARAMETERS,
   466→            target_column=target_column,
   467→            feature_file_name=feature_file_name,
   468→        ).train(symbol_list=symbol_list)
   469→
   470→    # Step 3: 生成权重
   471→    logger.info("[4H Pipeline] Step 3/4: Generating weights...")
   472→    generate_weights_4h()
   473→
   474→    # Step 4: 运行回测
   475→    logger.info("[4H Pipeline] Step 4/4: Running backtest...")
   476→    metrics = backtest_4h()
   477→
   478→    logger.info("=" * 60)
   479→    logger.info("[4H Pipeline] Pipeline completed!")
   480→    logger.info(f"[4H Pipeline] Final metrics: {metrics}")
   481→    logger.info("=" * 60)
   482→
   483→    return metrics
