     1→from typing import List, Optional

    18→    def __init__(
    19→        self,

    22→        hyper_parameters: dict = None,
    23→    ):
    24→        self.hyper_parameters = hyper_parameters
    25→        self.target_column = target_column

    27→        # @info 1. 读取特征数据
    28→        if not feature_file_name:
    29→            self.feature_data = get_local_feature_data()
    30→        else:
    31→            df = pd.read_csv(feature_file_name)
    32→            df["time"] = pd.to_datetime(df["time"])
    33→            self.feature_data = df
    34→            logger.info(
    35→                f"load feature from {feature_file_name} full feature start time: {df['time'].min()} end time: {df['time'].max()} "

    38→    def train(self, symbol_list: Optional[List[str]] = None):
    39→
    40→        # --- 2. 稳健的时间序列数据划分 (Train / Validation / Test) ---
    41→        logger.info("Splitting data into train, validation, and test sets...")
    42→        full_df = self.feature_data.sort_values("time")
    43→
    44→        if symbol_list:
    45→            full_df = full_df[full_df["symbol"].isin(symbol_list)]
    46→            logger.info(
    47→                f"Training on {len(symbol_list)} symbols, full_df size: {len(full_df)}"
    48→            )
    49→
    50→        # 测试集为最后90天
    51→        test_start_time = full_df["time"].max() - pd.Timedelta(days=90)
    52→        # 验证集为测试集之前的60天
    53→        val_start_time = test_start_time - pd.Timedelta(days=60)
    54→
    55→        # 验证集，用于 early stopping
    56→        val_df = full_df[
    57→            (full_df["time"] >= val_start_time) & (full_df["time"] < test_start_time)
    58→        ]
    59→        # 测试集，用于最终评估
    60→        test_df = full_df[full_df["time"] >= test_start_time]
    61→        # 最终训练集 = 所有在验证集之前的数据
    62→        train_df = full_df[full_df["time"] < val_start_time]
    63→
    64→        # 为了训练最强的模型，我们将训练集和验证集合并
    65→        # 但依然用独立的验证集进行早停
    66→        full_train_df = pd.concat([train_df, val_df])
    67→
    68→        logger.info(
    69→            f"Full training set size (for final model): {len(full_train_df)}, start time :{full_train_df['time'].min()}, end time: {full_train_df['time'].max()}"
    70→        )
    71→        logger.info(
    72→            f"Training set size (for early stopping): {len(train_df)}, start time :{train_df['time'].min()}, end time: {train_df['time'].max()}"
    73→        )
    74→        logger.info(
    75→            f"Validation set size (for early stopping): {len(val_df)}, start time :{val_df['time'].min()}, end time: {val_df['time'].max()}"
    76→        )
    77→        logger.info(
    78→            f"Test set size (for final evaluation): {len(test_df)}, start time :{test_df['time'].min()}, end time: {test_df['time'].max()}"
    79→        )
    80→
    81→        logger.info("Start training")
    82→
    83→        training_start_time = datetime.now()
    84→
    85→        X_train = train_df[self.feature_name]
    86→        y_train = train_df[self.target_column].astype(float)
    87→
    88→        X_val = val_df[self.feature_name]
    89→        y_val = val_df[self.target_column].astype(float)
    90→
    91→        train_data = lgb.Dataset(X_train, label=y_train)
    92→        val_data = lgb.Dataset(X_val, label=y_val, reference=train_data)
    93→
    94→        # optuna_num_boost_round = 182
    95→        # @info 加一个更多次数看看的效果
    96→        # optuna_num_boost_round = 500
    97→        # optuna_num_boost_round = 300
    98→        optuna_num_boost_round = 1000
    99→
   100→        num_boost_round = optuna_num_boost_round
   101→
   102→        logger.info(
   103→            f"Starting model training with number of boost {num_boost_round}, hyper parameters: {self.hyper_parameters}, total feature name :{len(RANKER_FEATURE_NAMES)}"
   104→        )
   105→        model = lgb.train(
   106→            self.hyper_parameters,
   107→            train_data,
   108→            num_boost_round=num_boost_round,
   109→            valid_sets=[
   110→                train_data,
   111→                val_data,
   112→            ],  # 传入训练集和验证集，方便观察是否存在过拟合
   113→            valid_names=["train", "valid"],  # 给它们起个名字
   114→            callbacks=[
   115→                lgb.log_evaluation(period=10),  # <--- 每 10 轮打印一次
   116→                # @info 早停，200轮没有提升就停止,保证和tune的一致
   117→                lgb.early_stopping(stopping_rounds=200, verbose=True),
   118→            ],
   119→        )
   120→
   121→        pkl_path = "regression_model_1h.pkl"
   122→        joblib.dump(model, pkl_path)
   123→        logger.info(f"Final model saved to {pkl_path}")
   124→
   125→        training_duration = datetime.now() - training_start_time
   126→        logger.info(f"Model training completed in {training_duration}")
   127→        logger.info(f"Best iteration found: {model.best_iteration}")  # 这一步至关重要
   128→
   129→        X_test = test_df[self.feature_name]
   130→
   131→        logger.info(f"Start generate prediction on test set of size {len(test_df)}")
   132→
   133→        # LightGBM multiclass 的 predict 默认返回 (n_sample , num_class) 的概率矩阵
   134→        preds = model.predict(X_test, num_iteration=model.best_iteration)
   135→        test_df["predicted_probability"] = preds
   136→        logger.info(
   137→            f"predicted_probability info :{test_df['predicted_probability'].describe()}"
   138→        )
   139→        save_path = "ranker_model_1h_predictions.feather"
   140→        test_df.to_feather(save_path)
   141→        logger.info(f"Predictions saved to {save_path}")

   143→    def generate_prediction(self, model: lgb.Booster, test_df: pd.DataFrame):
