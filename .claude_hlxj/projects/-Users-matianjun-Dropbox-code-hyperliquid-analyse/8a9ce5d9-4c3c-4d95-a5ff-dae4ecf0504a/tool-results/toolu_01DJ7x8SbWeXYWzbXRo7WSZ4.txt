    26→def preprocess_predict_df(predict_df: pd.DataFrame) -> pd.DataFrame:
    27→    # 获取第一个元素
    28→    first_elem = predict_df["all_symbol_weights"].iloc[0]
    29→
    30→    # 使用 isinstance 检查。
    31→    # 如果它是 dict, OrderedDict 或者任何继承自 dict 的类，这里都会返回 True。
    32→    # 我们加上 not，意味着只有当它 "不是" 字典家族的时候（比如是 str），才执行 eval。
    33→    if not isinstance(first_elem, dict):
    34→        predict_df["all_symbol_weights"] = predict_df.all_symbol_weights.apply(
    35→            lambda x: eval(x)
    36→        )
    37→
    38→    return predict_df
