    87→def compare_position_diff(target_df, actual_df, diff_threshold=1.0):
    88→    """
    89→    比较两个仓位 DataFrame 的差异。
    90→
    91→    Args:
    92→        target_df: 目标仓位 (ranker_position)
    93→        actual_df: 实际仓位 (lighter_account_position)
    94→        diff_threshold: 忽略小于此金额的差异 (默认为 1 USDT)
    95→
    96→    Returns:
    97→        summary_df: 包含所有差异的详细报表
    98→        missing_tokens: Lighter 账户缺少但 Ranker 有的币种
    99→        extra_tokens: Lighter 账户持有但 Ranker 没有的币种
   100→    """
   101→
   102→    # 1. 预处理：构造带符号的价值 (Long为正, Short为负) 以便准确计算敞口
   103→    def get_signed_value(row):
   104→        # 确保处理大小写敏感问题
   105→        side = row["side"].lower()
   106→        val = row["position_value"]
   107→        return val if side == "long" else -val
   108→
   109→    # 复制数据以防修改原始 DF
   110→    t_df = target_df.copy()
   111→    a_df = actual_df.copy()
   112→
   113→    t_df["signed_target"] = t_df.apply(get_signed_value, axis=1)
   114→    a_df["signed_actual"] = a_df.apply(get_signed_value, axis=1)
   115→
   116→    # 2. 合并数据 (Outer Join 以保留双方特有的 Token)
   117→    # 假设 symbol 列是主键 (如 "BTCUSDT")
   118→    merged = pd.merge(
   119→        t_df[["symbol", "signed_target"]],
   120→        a_df[["symbol", "signed_actual"]],
   121→        on="symbol",
   122→        how="outer",
   123→    ).fillna(0)
   124→
   125→    # 3. 计算差异 (实际 - 目标)
   126→    # Diff > 0 表示实际仓位比目标更"多" (或者空单更少)
   127→    # Diff < 0 表示实际仓位比目标更"少" (或者空单更多)
   128→    merged["diff_value"] = merged["signed_actual"] - merged["signed_target"]
   129→    merged["abs_diff"] = merged["diff_value"].abs()
   130→
   131→    # 4. 增加状态标记以便阅读
   132→    def get_status(row):
   133→        if row["signed_target"] != 0 and row["signed_actual"] == 0:
   134→            return "Missing in Account"  # Ranker有，账户没
   135→        elif row["signed_target"] == 0 and row["signed_actual"] != 0:
   136→            return "Extra in Account"  # Ranker没，账户有
   137→        elif row["abs_diff"] > diff_threshold:
   138→            return "Value Mismatch"  # 都有，但金额对不上
   139→        else:
   140→            return "Matched"  # 基本一致
   141→
   142→    merged["status"] = merged.apply(get_status, axis=1)
   143→
   144→    # 5. 整理输出格式
   145→    # 按差异绝对值降序排列，只保留有差异的行
   146→    result_df = merged[merged["status"] != "Matched"].sort_values(
   147→        "abs_diff", ascending=False
   148→    )
   149→
   150→    # 提取特定的列表
   151→    missing_tokens = result_df[result_df["status"] == "Missing in Account"][
   152→        "symbol"
   153→    ].tolist()
   154→    extra_tokens = result_df[result_df["status"] == "Extra in Account"][
   155→        "symbol"
   156→    ].tolist()
   157→
   158→    print(f"--- 差异概览 ---")
   159→    print(f"缺少 Token ({len(missing_tokens)}个): {missing_tokens}")
   160→    print(f"多余 Token ({len(extra_tokens)}个): {extra_tokens}")
   161→    print(f"\n--- 详细差异表 (Top 10) ---")
   162→    # 格式化一下显示，保留2位小数
   163→    pd.set_option("display.max_rows", 20)
   164→    pd.set_option("display.float_format", "{:.2f}".format)
   165→    print(
   166→        result_df[
   167→            ["symbol", "signed_target", "signed_actual", "diff_value", "status"]
   168→        ].head(10)
   169→    )
   170→    print(f"Diff token 总数: {len(result_df)}")
   171→
   172→    return result_df, missing_tokens, extra_tokens

   175→def audit_rebalance_logic(ranker_df, account_df, total_equity):
   176→    """
   177→    模拟 order_by_weights_and_total_value 的逻辑来审计仓位差异。
   178→
   179→    Args:
   180→        ranker_df: 包含 ['symbol', 'weigths', 'side']
   181→        account_df: 包含 ['symbol', 'position_value', 'side']
   182→        total_equity: 用于计算目标价值的总权益 (对应交易函数里的 total_value)
   183→    """
   184→
   185→    # -------------------------------------------------
   186→    # 1. 模拟交易代码的权重归一化 (Step 1 of your function)
   187→    # -------------------------------------------------
   188→    ranker = ranker_df.copy()
   189→
   190→    # 确保 weight 也是带符号的，或者根据 side 处理
   191→    # 假设你的 ranker_df['weigths'] 可能是正数，通过 side='short' 变负
   192→    def get_signed_weight(row):
   193→        w = abs(row["weigths"])
   194→        return w if row["side"] == "long" else -w
   195→
   196→    ranker["raw_signed_weight"] = ranker.apply(get_signed_weight, axis=1)
   197→
   198→    # 计算权重绝对值之和
   199→    weights_sum = ranker["raw_signed_weight"].abs().sum()
   200→
   201→    # 执行归一化 (关键步骤!)
   202→    if weights_sum == 0:
   203→        ranker["normalized_weight"] = 0
   204→    else:
   205→        ranker["normalized_weight"] = ranker["raw_signed_weight"] / weights_sum
   206→
   207→    # 计算目标价值 (Target Value)
   208→    ranker["target_value"] = ranker["normalized_weight"] * total_equity
   209→
   210→    # -------------------------------------------------
   211→    # 2. 准备账户数据
   212→    # -------------------------------------------------
   213→    account = account_df.copy()
   214→
   215→    def get_signed_position(row):
   216→        val = abs(row["position_value"])
   217→        return val if row["side"] == "long" else -val
   218→
   219→    account["actual_value"] = account.apply(get_signed_position, axis=1)
   220→
   221→    # -------------------------------------------------
   222→    # 3. 合并与比较
   223→    # -------------------------------------------------
   224→    merged = pd.merge(
   225→        ranker[["symbol", "target_value"]],
   226→        account[["symbol", "actual_value"]],
   227→        on="symbol",
   228→        how="outer",
   229→    ).fillna(0)
   230→
   231→    # 计算执行层面的 Diff (Target - Actual)
   232→    # 正数 = 需要买入 (Long More / Close Short)
   233→    # 负数 = 需要卖出 (Short More / Close Long)
   234→    merged["trade_diff"] = merged["target_value"] - merged["actual_value"]
   235→
   236→    # 增加一列：模拟交易代码的操作类型
   237→    def classify_action(row):
   238→        t = row["target_value"]
   239→        a = row["actual_value"]
   240→        diff = row["trade_diff"]
   241→
   242→        if abs(diff) < 1.0:  # 忽略微小金额
   243→            return "No Action"
   244→
   245→        # 异号 (翻转)
   246→        if (t > 0 and a < 0) or (t < 0 and a > 0):
   247→            return "FLIP (Close & Open)"
   248→
   249→        # 目标为0 (平仓)
   250→        if t == 0 and a != 0:
   251→            return "Close Position"
   252→
   253→        # 同号但大小不同
   254→        if abs(diff) > 0:
   255→            if t > 0:  # 多头端
   256→                return "Add Long" if t > a else "Reduce Long"
   257→            else:  # 空头端
   258→                return "Add Short" if t < a else "Reduce Short"
   259→
   260→        return "Stable"
   261→
   262→    merged["action_type"] = merged.apply(classify_action, axis=1)
   263→
   264→    # 排序：按交易金额大小排序
   265→    merged["abs_trade_size"] = merged["trade_diff"].abs()
   266→    result = merged.sort_values("abs_trade_size", ascending=False)
   267→
   268→    return result

   525→@app.command("executor")
   526→def start_executor_process(
   527→    account_id: int = typer.Argument(..., help="Account ID for the executor process"),
   528→    exchange_account: str = typer.Option(
   529→        "main", "--exchange", "-e", help="Exchange account name (e.g., main, sub)"
   530→    ),
   531→):
   532→    """
   533→    Start the Executor process (Process B) for distributed trading.
   534→
   535→    The Executor process handles:
