    48→def my_round(num, n) -> float:
    49→    """
    50→    Returns a value rounded down to a specific number of decimal places.
    51→
    52→    Args:
    53→        num: the number you want to round
    54→        n: the decimal precision
    55→
    56→    Returns:
    57→
    58→    """
    59→    # history reason to have this code
    60→    # TODO: delete this code and find out where the num is not a float
    61→    if isinstance(num, str):
    62→        num = float(num)
    63→
    64→    if not isinstance(n, int):
    65→        raise TypeError("decimal places must be an integer")
    66→    elif n < 0:
    67→        raise ValueError("decimal places has to be 0 or more")
    68→    elif n == 0:
    69→        return math.floor(num)
    70→
    71→    factor = 10**n
    72→    return math.floor(num * factor) / factor

   110→    async def get_kline(
   111→        self,
   112→        symbol: str,
   113→        timeframe: str,
   114→        start_date: str,
   115→        end_date: str = None,
   116→        to_db: bool = False,
   117→        to_local: bool = False,
   118→        limit=None,
   119→    ) -> pd.DataFrame:
   120→        """
   121→        Args:
   122→            to_db: 是否入库，默认不入库
   123→            timeframe: '1m','5m','15m','30m','1h','2h','4h','6h','12h','1d'
   124→            start_date: K线起始时间，"%Y-%m-%d %H:%M:%S"
   125→            end_date: K线结束时间，"%Y-%m-%d %H:%M:%S"
   126→        Returns:[
   127→        ]
   128→        """
   129→        granularity = self.parse_time_frame(timeframe) * 1000
   130→        startTime = int(arrow.get(start_date).timestamp() * 1000)
   131→        end = int(arrow.get(end_date).timestamp() * 1000)
   132→        if not limit:
   133→            limit = 1500
   134→        path = f"{self.get_url(self.MarketType.USDT_FUTURE)}/v1/klines"
   135→        data = []
   136→        while startTime < end:
   137→            endTime = min(startTime + granularity * limit, end)
   138→            param = {
   139→                "symbol": symbol,
   140→                "startTime": startTime,
   141→                "endTime": endTime,
   142→                "interval": timeframe,
   143→                "limit": limit,
   144→            }
   145→            self.logger.info(
   146→                f"正在获取{symbol}-{timeframe.upper()} K线数据:{datetime.fromtimestamp(startTime / 1000, tz=utc)} - {datetime.fromtimestamp(endTime / 1000, tz=utc)}"
   147→            )
   148→            startTime = endTime
   149→            for i in range(2):
   150→                try:
   151→                    res = await self.public_request_get(path, data=param)
   152→                    data.extend(res)
   153→                    break
   154→                except Exception as e:
   155→                    self.logger.error(f"获取K线异常:{e} 开始第{i}次重试", exc_info=True)
   156→                    await asyncio.sleep(10)
   157→
   158→            # 1 / (2400 /60) = 0.025
   159→            await asyncio.sleep(0.02)
   160→
   161→        df = pd.DataFrame(data)
   162→        df = df[[0, 1, 2, 3, 4, 7]]
   163→        df[7] = round(df[7].astype(float) / 10000, 1)
   164→
   165→        df.columns = ["candle_begin_time", "open", "high", "low", "close", "volume"]
   166→        df["candle_begin_time"] = pd.to_datetime(df["candle_begin_time"], unit="ms")
   167→        df.drop_duplicates("candle_begin_time", inplace=True)
   168→        df.sort_values(["candle_begin_time"], inplace=True)
   169→        self.logger.info(
   170→            f"获取{symbol}-{timeframe.upper()} K线数据完毕，共计{len(df)}条记录"
   171→        )
   172→        return df

   174→    @classmethod
   175→    async def get_symbols(cls, market_type: str, to_db: bool = True):
   176→        """获取所有的交易对
   177→
   178→        Args:
   179→            market_type:  市场类型
   180→            to_db:  是否把结果存入数据库
   181→
   182→        Returns:
   183→
   184→        """
   185→        path = f"{cls.get_url(market_type)}/v1/exchangeInfo"
   186→        data = (await cls.public_request_get(path))["symbols"]
   187→        symbols = {}
   188→        for d in data:
   189→            try:
   190→                symbol = d["symbol"]
   191→                amount_precision = 0
   192→                price_precision = 0
   193→                min_amount = 0
   194→                min_cost = 0
   195→
   196→                max_amount = 0
   197→
   198→                note = ""
   199→                if market_type == cls.MarketType.SPOT:
   200→                    for x in d["filters"]:
   201→                        if x["filterType"] == "LOT_SIZE":
   202→                            amount_precision = (
   203→                                -Decimal(str(float(x["stepSize"]))).as_tuple().exponent
   204→                            )
   205→                            min_amount = float(x["stepSize"])
   206→
   207→                        if x["filterType"] == "PRICE_FILTER":
   208→                            price_precision = (
   209→                                -Decimal(str(float(x["tickSize"]))).as_tuple().exponent
   210→                            )
   211→
   212→                        if x["filterType"] == "MIN_NOTIONAL":
   213→                            min_cost = int(float(x["minNotional"]))
   214→
   215→                elif market_type == cls.MarketType.USDT_FUTURE:
   216→                    for x in d["filters"]:
   217→                        if x["filterType"] == "MARKET_LOT_SIZE":
   218→                            min_amount = float(x["minQty"])
   219→                            max_amount = float(x["maxQty"])
   220→                        elif x["filterType"] == "PRICE_FILTER":
   221→                            note = x["tickSize"]
   222→
   223→                    amount_precision = d.get("quantityPrecision")
   224→                    price_precision = d.get("pricePrecision")
   225→
   226→                symbols[symbol] = {
   227→                    "symbol": symbol,
   228→                    "underlying": f"{d['baseAsset']}{d['quoteAsset']}",
   229→                    "exchange": cls.EXCHANGE,
   230→                    "market_type": market_type,
   231→                    "contract_val": (
   232→                        d["contractSize"]
   233→                        if market_type == cls.MarketType.COIN_FUTURE
   234→                        else 1
   235→                    ),
   236→                    "is_coin_base": True,
   237→                    "is_tradable": True,
   238→                    "amount_precision": amount_precision,
   239→                    "price_precision": price_precision,
   240→                    "min_amount": min_amount,
   241→                    "max_amount": max_amount,
   242→                    "min_cost": min_cost,
   243→                    "category": 0,
   244→                    "volume": 0,
   245→                    "base_coin": d.get("baseAsset"),
   246→                    "quote_coin": d.get("quoteAsset"),
   247→                    "note": note,
   248→                }
   249→            except Exception as e:
   250→                cls.logger.info(d)
   251→                cls.logger.info(e)
   252→
   253→        return symbols
