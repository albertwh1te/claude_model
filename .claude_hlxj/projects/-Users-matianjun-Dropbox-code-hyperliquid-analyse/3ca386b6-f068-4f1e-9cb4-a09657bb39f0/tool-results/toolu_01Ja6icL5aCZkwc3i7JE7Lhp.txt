    89→def compare_position_diff(target_df, actual_df, diff_threshold=1.0):
    90→    """
    91→    比较两个仓位 DataFrame 的差异。
    92→
    93→    Args:
    94→        target_df: 目标仓位 (ranker_position)
    95→        actual_df: 实际仓位 (lighter_account_position)
    96→        diff_threshold: 忽略小于此金额的差异 (默认为 1 USDT)
    97→
    98→    Returns:
    99→        summary_df: 包含所有差异的详细报表
   100→        missing_tokens: Lighter 账户缺少但 Ranker 有的币种
   101→        extra_tokens: Lighter 账户持有但 Ranker 没有的币种
   102→    """
   103→
   104→    # 1. 预处理：构造带符号的价值 (Long为正, Short为负) 以便准确计算敞口
   105→    def get_signed_value(row):
   106→        # 确保处理大小写敏感问题
   107→        side = row["side"].lower()
   108→        val = row["position_value"]
   109→        return val if side == "long" else -val
   110→
   111→    # 复制数据以防修改原始 DF
   112→    t_df = target_df.copy()
   113→    a_df = actual_df.copy()
   114→
   115→    t_df["signed_target"] = t_df.apply(get_signed_value, axis=1)
   116→    a_df["signed_actual"] = a_df.apply(get_signed_value, axis=1)
   117→
   118→    # 2. 合并数据 (Outer Join 以保留双方特有的 Token)
   119→    # 假设 symbol 列是主键 (如 "BTCUSDT")
   120→    merged = pd.merge(
   121→        t_df[["symbol", "signed_target"]],
   122→        a_df[["symbol", "signed_actual"]],
   123→        on="symbol",
   124→        how="outer",
   125→    ).fillna(0)
   126→
   127→    # 3. 计算差异 (实际 - 目标)
   128→    # Diff > 0 表示实际仓位比目标更"多" (或者空单更少)
   129→    # Diff < 0 表示实际仓位比目标更"少" (或者空单更多)
   130→    merged["diff_value"] = merged["signed_actual"] - merged["signed_target"]
   131→    merged["abs_diff"] = merged["diff_value"].abs()
   132→
   133→    # 4. 增加状态标记以便阅读
   134→    def get_status(row):
   135→        if row["signed_target"] != 0 and row["signed_actual"] == 0:
   136→            return "Missing in Account"  # Ranker有，账户没
   137→        elif row["signed_target"] == 0 and row["signed_actual"] != 0:
   138→            return "Extra in Account"  # Ranker没，账户有
   139→        elif row["abs_diff"] > diff_threshold:
   140→            return "Value Mismatch"  # 都有，但金额对不上
   141→        else:
   142→            return "Matched"  # 基本一致
   143→
   144→    merged["status"] = merged.apply(get_status, axis=1)
   145→
   146→    # 5. 整理输出格式
   147→    # 按差异绝对值降序排列，只保留有差异的行
   148→    result_df = merged[merged["status"] != "Matched"].sort_values(
   149→        "abs_diff", ascending=False
   150→    )
   151→
   152→    # 提取特定的列表
   153→    missing_tokens = result_df[result_df["status"] == "Missing in Account"][
   154→        "symbol"
   155→    ].tolist()
   156→    extra_tokens = result_df[result_df["status"] == "Extra in Account"][
   157→        "symbol"
   158→    ].tolist()
   159→
   160→    print(f"--- 差异概览 ---")
   161→    print(f"缺少 Token ({len(missing_tokens)}个): {missing_tokens}")
   162→    print(f"多余 Token ({len(extra_tokens)}个): {extra_tokens}")
   163→    print(f"\n--- 详细差异表 (Top 10) ---")
   164→    # 格式化一下显示，保留2位小数
   165→    pd.set_option("display.max_rows", 20)
   166→    pd.set_option("display.float_format", "{:.2f}".format)
   167→    print(
   168→        result_df[
   169→            ["symbol", "signed_target", "signed_actual", "diff_value", "status"]
   170→        ].head(10)
   171→    )
   172→    print(f"Diff token 总数: {len(result_df)}")
   173→
   174→    return result_df, missing_tokens, extra_tokens

   177→def audit_rebalance_logic(ranker_df, account_df, total_equity):
   178→    """
   179→    模拟 order_by_weights_and_total_value 的逻辑来审计仓位差异。
   180→
   181→    Args:
   182→        ranker_df: 包含 ['symbol', 'weigths', 'side']
   183→        account_df: 包含 ['symbol', 'position_value', 'side']
   184→        total_equity: 用于计算目标价值的总权益 (对应交易函数里的 total_value)
   185→    """
   186→
   187→    # -------------------------------------------------
   188→    # 1. 模拟交易代码的权重归一化 (Step 1 of your function)
   189→    # -------------------------------------------------
   190→    ranker = ranker_df.copy()
   191→
   192→    # 确保 weight 也是带符号的，或者根据 side 处理
   193→    # 假设你的 ranker_df['weigths'] 可能是正数，通过 side='short' 变负
   194→    def get_signed_weight(row):
   195→        w = abs(row["weigths"])
   196→        return w if row["side"] == "long" else -w
   197→
   198→    ranker["raw_signed_weight"] = ranker.apply(get_signed_weight, axis=1)
   199→
   200→    # 计算权重绝对值之和
   201→    weights_sum = ranker["raw_signed_weight"].abs().sum()
   202→
   203→    # 执行归一化 (关键步骤!)
   204→    if weights_sum == 0:
   205→        ranker["normalized_weight"] = 0
   206→    else:
   207→        ranker["normalized_weight"] = ranker["raw_signed_weight"] / weights_sum
   208→
   209→    # 计算目标价值 (Target Value)
   210→    ranker["target_value"] = ranker["normalized_weight"] * total_equity
   211→
   212→    # -------------------------------------------------
   213→    # 2. 准备账户数据
   214→    # -------------------------------------------------
   215→    account = account_df.copy()
   216→
   217→    def get_signed_position(row):
   218→        val = abs(row["position_value"])
   219→        return val if row["side"] == "long" else -val
   220→
   221→    account["actual_value"] = account.apply(get_signed_position, axis=1)
   222→
   223→    # -------------------------------------------------
   224→    # 3. 合并与比较
   225→    # -------------------------------------------------
   226→    merged = pd.merge(
   227→        ranker[["symbol", "target_value"]],
   228→        account[["symbol", "actual_value"]],
   229→        on="symbol",
   230→        how="outer",
   231→    ).fillna(0)
   232→
   233→    # 计算执行层面的 Diff (Target - Actual)
   234→    # 正数 = 需要买入 (Long More / Close Short)
   235→    # 负数 = 需要卖出 (Short More / Close Long)
   236→    merged["trade_diff"] = merged["target_value"] - merged["actual_value"]
   237→
   238→    # 增加一列：模拟交易代码的操作类型
   239→    def classify_action(row):
   240→        t = row["target_value"]
   241→        a = row["actual_value"]
   242→        diff = row["trade_diff"]
   243→
   244→        if abs(diff) < 1.0:  # 忽略微小金额
   245→            return "No Action"
   246→
   247→        # 异号 (翻转)
   248→        if (t > 0 and a < 0) or (t < 0 and a > 0):
   249→            return "FLIP (Close & Open)"
   250→
   251→        # 目标为0 (平仓)
   252→        if t == 0 and a != 0:
   253→            return "Close Position"
   254→
   255→        # 同号但大小不同
   256→        if abs(diff) > 0:
   257→            if t > 0:  # 多头端
   258→                return "Add Long" if t > a else "Reduce Long"
   259→            else:  # 空头端
   260→                return "Add Short" if t < a else "Reduce Short"
   261→
   262→        return "Stable"
   263→
   264→    merged["action_type"] = merged.apply(classify_action, axis=1)
   265→
   266→    # 排序：按交易金额大小排序
   267→    merged["abs_trade_size"] = merged["trade_diff"].abs()
   268→    result = merged.sort_values("abs_trade_size", ascending=False)
   269→
   270→    return result
