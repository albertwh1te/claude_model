   164→def _serialize_bytes_to_hex(obj: Any) -> Any:
   165→    """Recursively convert bytes to hex strings for JSON serialization."""
   166→    if isinstance(obj, bytes):
   167→        return "0x" + obj.hex()
   168→    elif isinstance(obj, dict):
   169→        return {k: _serialize_bytes_to_hex(v) for k, v in obj.items()}
   170→    elif isinstance(obj, list):
   171→        return [_serialize_bytes_to_hex(v) for v in obj]
   172→    return obj

   233→# ==================== Utility Functions ====================
   234→
   235→
   236→def hex_to_bytes32(hex_str: str) -> bytes:
   237→    """Convert hex string to bytes32."""
   238→    if hex_str.startswith("0x"):
   239→        hex_str = hex_str[2:]
   240→    return bytes.fromhex(hex_str.zfill(64))
   241→
   242→
   243→def bytes32_to_hex(b: bytes) -> str:
   244→    """Convert bytes32 to hex string."""
   245→    return "0x" + b.hex()
   246→
   247→
   248→def subaccount_to_bytes32(subaccount: Subaccount) -> bytes:
   249→    """Convert subaccount to bytes32."""
   250→    if isinstance(subaccount, bytes):
   251→        return subaccount
   252→    if isinstance(subaccount, str):
   253→        return hex_to_bytes32(subaccount)
   254→    if isinstance(subaccount, SubaccountParams):
   255→        owner = subaccount.subaccount_owner or ""
   256→        name = subaccount.subaccount_name or "default"
   257→        # Encode: owner address (20 bytes) + name (12 bytes)
   258→        owner_bytes = bytes.fromhex(owner[2:] if owner.startswith("0x") else owner)
   259→        name_bytes = name.encode("utf-8")[:12].ljust(12, b"\x00")
   260→        return owner_bytes + name_bytes
   261→    raise ValueError(f"Cannot convert {type(subaccount)} to bytes32")
   262→
   263→
   264→def subaccount_to_hex(subaccount: Subaccount) -> str:
   265→    """Convert subaccount to hex string."""
   266→    return bytes32_to_hex(subaccount_to_bytes32(subaccount))

   269→def gen_order_nonce(recv_time_ms: Optional[int] = None) -> int:
   270→    """Generate order nonce."""
   271→    if recv_time_ms is None:
   272→        recv_time_ms = int(time.time() * 1000)
   273→    random_bits = secrets.randbits(32)
   274→    return (recv_time_ms << 20) | (random_bits & 0xFFFFF)
   275→
   276→
   277→def get_expiration_timestamp(seconds_from_now: int) -> int:
   278→    """Get expiration timestamp."""
   279→    return int(time.time()) + seconds_from_now

   282→def gen_order_verifying_contract(product_id: int) -> str:
   283→    """
   284→    Generates the order verifying contract address based on the product ID.
   285→
   286→    Args:
   287→        product_id (int): The product ID for which to generate the verifying contract address.
   288→
   289→    Returns:
   290→        str: The generated order verifying contract address in hexadecimal format.
   291→    """
   292→    be_bytes = product_id.to_bytes(20, byteorder="big", signed=False)
   293→    return "0x" + be_bytes.hex()

   324→def round_x18(value: int, increment: int) -> int:
   325→    """Round x18 value to increment."""
   326→    if increment == 0:
   327→        return value
   328→    return (value // increment) * increment

   331→# ==================== Order Utilities ====================
   332→
   333→
   334→ORDER_APPENDIX_VERSION = 1
   335→
   336→
   337→def build_appendix(
   338→    order_type: OrderType = OrderType.DEFAULT,
   339→    reduce_only: bool = False,
   340→    isolated: bool = False,
   341→    isolated_margin: int = 0,
   342→    trigger_type: OrderAppendixTriggerType = OrderAppendixTriggerType.NO_TRIGGER,
   343→    twap_times: int = 0,
   344→    twap_slippage_frac: float = 0,
   345→) -> int:
   346→    """Build order appendix integer."""
   347→    order_type_map = {
   348→        OrderType.DEFAULT: 0,
   349→        OrderType.IOC: 1,
   350→        OrderType.FOK: 2,
   351→        OrderType.POST_ONLY: 3,
   352→    }
   353→
   354→    appendix = ORDER_APPENDIX_VERSION
   355→    appendix |= (1 if isolated else 0) << 8
   356→    appendix |= order_type_map.get(order_type, 0) << 9
   357→    appendix |= (1 if reduce_only else 0) << 11
   358→    appendix |= trigger_type.value << 12
   359→
   360→    if trigger_type in (
   361→        OrderAppendixTriggerType.TWAP,
   362→        OrderAppendixTriggerType.TWAP_CUSTOM_AMOUNTS,
   363→    ):
   364→        twap_slippage_encoded = int(twap_slippage_frac * 1000) & 0x3FF
   365→        value = (twap_times & 0x1FF) | (twap_slippage_encoded << 9)
   366→        appendix |= value << 64
   367→    elif isolated and isolated_margin > 0:
   368→        appendix |= (isolated_margin & ((1 << 64) - 1)) << 64
   369→
   370→    return appendix

  1408→def to_trigger_execute_request(params: TriggerExecuteParams) -> TriggerExecuteRequest:
  1409→    """
  1410→    Maps TriggerExecuteParams to its corresponding TriggerExecuteRequest object.
  1411→
  1412→    Args:
  1413→        params (TriggerExecuteParams): The parameters to be executed.
  1414→
  1415→    Returns:
  1416→        TriggerExecuteRequest: The corresponding TriggerExecuteRequest object.
  1417→    """
  1418→    if isinstance(params, PlaceTriggerOrderParams):
  1419→        return PlaceTriggerOrderRequest(place_order=params)
  1420→    elif isinstance(params, PlaceTriggerOrdersParams):
