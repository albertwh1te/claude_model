   415→                api.get_kline(
   416→                    # timeframe="1h",
   417→                    timeframe=kline_info_pair["kline_name"],

   419→                    end_date=end_time,
   420→                    symbol=symbol,

   423→            bar.set_description(f"insert {symbol}")
