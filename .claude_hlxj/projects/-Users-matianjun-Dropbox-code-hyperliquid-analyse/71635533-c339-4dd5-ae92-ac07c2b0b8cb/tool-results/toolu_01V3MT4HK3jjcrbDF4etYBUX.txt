   220→    async def place_twap_order(
   221→        self,
   222→        product_id: int,
   223→        price_x18: str,
   224→        total_amount_x18: str,
   225→        times: int,
   226→        slippage_frac: float,
   227→        interval_seconds: int,
   228→        sender: Optional[str] = None,
   229→        subaccount_owner: Optional[str] = None,
   230→        subaccount_name: str = "default",
   231→        expiration: Optional[int] = None,
   232→        nonce: Optional[int] = None,
   233→        custom_amounts_x18: Optional[List[str]] = None,
   234→        reduce_only: bool = False,
   235→        spot_leverage: Optional[bool] = None,
   236→        id: Optional[int] = None,
   237→    ) -> ExecuteResponse:
   238→        """
   239→        Place a TWAP (Time-Weighted Average Price) order.
   240→
   241→        Args:
   242→            product_id: The product ID for the order.
   243→            price_x18: The limit price multiplied by 1e18.
   244→            total_amount_x18: The total amount to trade (signed, negative for sell).
   245→            times: Number of TWAP executions (1-500).
   246→            slippage_frac: Slippage tolerance as a fraction (e.g., 0.01 for 1%).
   247→            interval_seconds: Time interval between executions in seconds.
   248→            sender: The sender address. If provided, takes precedence.
   249→            subaccount_owner: The subaccount owner address.
   250→            subaccount_name: The subaccount name. Defaults to "default".
   251→            expiration: Order expiration timestamp.
   252→            nonce: Order nonce.
   253→            custom_amounts_x18: Custom amounts for each execution.
   254→            reduce_only: Whether this is a reduce-only order.
   255→            spot_leverage: Whether to use spot leverage.
   256→            id: Optional order ID.
   257→
   258→        Returns:
   259→            ExecuteResponse: The response from placing the TWAP order.
   260→
   261→        Raises:
   262→            MissingTriggerClient: If trigger client is not configured.
   263→        """
   264→        if self.context.trigger_client is None:
   265→            raise MissingTriggerClient()
   266→        return await self.context.trigger_client.place_twap_order(
   267→            product_id=product_id,
   268→            price_x18=price_x18,
   269→            total_amount_x18=total_amount_x18,
   270→            times=times,
   271→            slippage_frac=slippage_frac,
   272→            interval_seconds=interval_seconds,
   273→            sender=sender,
   274→            subaccount_owner=subaccount_owner,
   275→            subaccount_name=subaccount_name,
   276→            expiration=expiration,
   277→            nonce=nonce,
   278→            custom_amounts_x18=custom_amounts_x18,
   279→            reduce_only=reduce_only,
   280→            spot_leverage=spot_leverage,
   281→            id=id,
   282→        )

   284→    async def place_price_trigger_order(
   285→        self,
   286→        product_id: int,
   287→        price_x18: str,
   288→        amount_x18: str,
   289→        trigger_price_x18: str,
   290→        trigger_type: str,
   291→        sender: Optional[str] = None,
   292→        subaccount_owner: Optional[str] = None,
   293→        subaccount_name: str = "default",
   294→        expiration: Optional[int] = None,
   295→        nonce: Optional[int] = None,
   296→        reduce_only: bool = False,
   297→        order_type: OrderType = OrderType.DEFAULT,
   298→        spot_leverage: Optional[bool] = None,
   299→        id: Optional[int] = None,
   300→        dependency: Optional[dict] = None,
   301→    ) -> ExecuteResponse:
   302→        """
   303→        Place a price trigger order.
   304→
   305→        Args:
   306→            product_id: The product ID for the order.
   307→            price_x18: The limit price multiplied by 1e18.
   308→            amount_x18: The amount to trade (signed, negative for sell).
   309→            trigger_price_x18: The trigger price multiplied by 1e18.
   310→            trigger_type: Type of price trigger.
   311→            sender: The sender address.
   312→            subaccount_owner: The subaccount owner address.
   313→            subaccount_name: The subaccount name. Defaults to "default".
   314→            expiration: Order expiration timestamp.
   315→            nonce: Order nonce.
   316→            reduce_only: Whether this is a reduce-only order.
   317→            order_type: Order execution type.
   318→            spot_leverage: Whether to use spot leverage.
   319→            id: Optional order ID.
   320→            dependency: Optional dependency trigger dict.
   321→
   322→        Returns:
   323→            ExecuteResponse: The response from placing the price trigger order.
   324→
   325→        Raises:
   326→            MissingTriggerClient: If trigger client is not configured.
   327→        """
   328→        if self.context.trigger_client is None:
   329→            raise MissingTriggerClient()
   330→
   331→        from nado_api.types import Dependency
   332→
   333→        dependency_obj = None
   334→        if dependency is not None:
   335→            dependency_obj = Dependency.parse_obj(dependency)
   336→
   337→        return await self.context.trigger_client.place_price_trigger_order(
   338→            product_id=product_id,
   339→            price_x18=price_x18,
   340→            amount_x18=amount_x18,
   341→            trigger_price_x18=trigger_price_x18,
   342→            trigger_type=trigger_type,
   343→            sender=sender,
   344→            subaccount_owner=subaccount_owner,
   345→            subaccount_name=subaccount_name,
   346→            expiration=expiration,
   347→            nonce=nonce,
   348→            reduce_only=reduce_only,
   349→            order_type=order_type,
   350→            spot_leverage=spot_leverage,
   351→            id=id,
   352→            dependency=dependency_obj,
   353→        )
